# if-else
`if (x > 10) { ... } else { ... }`
# if- else if -else
`if (x == 1) { ... } else if( x == 2) { ... } else { ... }`
# switch-case
`switch (x) { case 1: ... case 2: ... default: ... }`
# switch-case (Ranges/Conditions)
`switch (x) { case 0..10: ... case isEven(x): ... default: ... }`
# while loop
`while ( x < 10) { ... }`
# do-while loop
`do { ... } while (x < 10);`
# for loop
`for (let i = 0; i < 10; i++) { ... }`
# for loop (range)
`for i = 0 to 10 { ... }`
# for-in loop
`for key in obj { ... }`
# for-of loop
`for value in arr { ... }`
# forEach (array)
`arr.forEach(element => { ... })`
# break
`for i in 0..10 { if i == 5 { break } }`
# continue
`for i in 0..10 { if i == 5 { continue } }`
# return
`return x`
